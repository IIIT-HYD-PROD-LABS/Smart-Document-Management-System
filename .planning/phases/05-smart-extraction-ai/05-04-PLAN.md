   ```typescript
   function ExtractionConfidenceBadge({ confidence }: { confidence: number }) {
       const pct = Math.round(confidence * 100);
       let color: string;
       let bg: string;
       if (confidence >= 0.8) { color = "text-[#10b981]"; bg = "bg-[#10b981]/10"; }
       else if (confidence >= 0.5) { color = "text-[#f59e0b]"; bg = "bg-[#f59e0b]/10"; }
       else { color = "text-[#ef4444]"; bg = "bg-[#ef4444]/10"; }
       return (
           <span className={`text-[10px] px-1.5 py-0.5 rounded ${bg} ${color} font-medium`}>
               {pct}%
           </span>
       );
   }
   ```

   c) Create an `ExtractionStatusBadge` component (inline):

   - null/undefined: don't render anything

   ```tsx
   {doc.ai_summary && (
   ```

   ```tsx
           </div>
       </div>
   )}
   ```

   f) Create a helper function `renderFieldGroup`:

   ```typescript
   function renderFieldGroup(title: string, fields?: Array<{ value: string | number | null; confidence: number; source_hint?: string | null }>) {
       if (!fields || fields.length === 0) return null;
       return (
           <div>
               <p className="text-[11px] text-[#52525b] uppercase tracking-wider mb-2">{title}</p>
               <div className="flex flex-wrap gap-2">
                   {fields.map((field, i) => (
                       <div key={i} className="flex items-center gap-2 bg-[#18181b] border border-[#27272a] rounded-md px-3 py-1.5">
                           <span className="text-sm text-white">{String(field.value ?? "-")}</span>
                           <ExtractionConfidenceBadge confidence={field.confidence} />
                       </div>
                   ))}
               </div>
           </div>
       );
   }
   ```
