1. Create `backend/app/routers/ml.py` with a GET /api/ml/evaluation endpoint:
   ```python
   import json
   from pathlib import Path
   from fastapi import APIRouter, Depends, HTTPException
   import structlog
   from app.config import settings
   from app.utils.security import get_current_user

   logger = structlog.stdlib.get_logger()
   router = APIRouter(prefix="/api/ml", tags=["ML"])

   @router.get("/evaluation")
   def get_evaluation_report(current_user=Depends(get_current_user)):
       eval_path = Path(settings.MODEL_DIR) / "evaluation" / "evaluation_report.json"
       if not eval_path.exists():
           raise HTTPException(status_code=404, detail="No evaluation report available")
       with open(eval_path) as f:
           report = json.load(f)
       return report
   ```

   function ConfidenceBadge({ score }: { score: number }) {
     if (score <= 0) return null;
     const pct = Math.round(score * 100);
     let colorClass: string;
     let label: string;
     if (score >= 0.8) {
       colorClass = "bg-[#10b981]/10 text-[#10b981]";
       label = "High";
     } else if (score >= 0.5) {
       colorClass = "bg-[#f59e0b]/10 text-[#f59e0b]";
       label = "Medium";
     } else {
       colorClass = "bg-[#ef4444]/10 text-[#ef4444]";
       label = "Low";
     }
     return (
       <span className={`text-[11px] px-2 py-0.5 rounded ${colorClass}`} title={`${label} confidence`}>
         {pct}%
       </span>
     );
   }
